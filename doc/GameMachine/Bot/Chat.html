def schedule_activities
  sleep(rand(10))
  duration = JavaLib::Duration.create(rand(10) + 1, java.util.concurrent.TimeUnit::SECONDS)
  @scheduler.schedule(duration, duration, get_self, "send_chat_message", @dispatcher, nil)
end
