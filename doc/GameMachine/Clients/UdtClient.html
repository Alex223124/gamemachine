def connect
  address = JavaLib::InetSocketAddress.new(@host, @port)
  @socket = JavaLib::SocketUDT.new(JavaLib::TypeUDT::DATAGRAM)
  @socket.setBlocking(true)
  @socket.connect(address)
  @socket
end
