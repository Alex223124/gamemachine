def add_bucket(bucket)
  unless @buckets.include?(bucket)
    @ring << bucket
    @buckets << bucket
  end
end

def remove_bucket(bucket)
  if @buckets.include?(bucket)
    @buckets.delete_if {|b| b == bucket}
    @ring.delete(bucket)
  end
end
