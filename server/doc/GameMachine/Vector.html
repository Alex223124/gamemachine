def distance_squared(v)
  dx = @x - v.x
  dy = @y - v.y
  dz = @z - v.z
  (dx * dx + dy * dy + dz * dz);
end

def interpolate(vector,change_amount,use_z=false)
  @x=(1-change_amount)*@x + change_amount*vector.x
  @y=(1-change_amount)*@y + change_amount*vector.y
  if use_z
    @z=(1-change_amount)*@z + change_amount*vector.z
  end
end
